112 public void SetSeed(int seed)
113 {
114 Seed = seed;
115 Position1 = 0;
116 Position2 = 21;
117 var mj = seed == int.MinValue ? int.MaxValue : Math.Abs(seed);
118 mj = 161803398 - mj;
119 Seeds[55] = mj;
120 var mk = 1;
121
122 // Populate random seed array.
123 for (int i = 1; i < 55; i++)
124 {
125 var index = 21 * i % 55;
126 Seeds[index] = mk;
127 mk = Mod(mj - mk);
128 mj = Seeds[index];
129 }
130
131 // Generate random numbers to reduce seed bias.
132 for (int k = 1; k < 5; k++)
133 {
134 for (int i = 1; i < 56; i++)
135 {
136 Seeds[i] = Mod(Seeds[i] - Seeds[1 + (i + 30) % 55]);
137 }

114 public void SetSeed(int seed)
115 {
116 Seed = seed;
117 Position1 = 0;
118 Position2 = 21;
119 var mj = seed == int.MinValue ? int.MaxValue : Math.Abs(seed);
120 mj = 161803398 - mj;
121 Seeds[55] = mj;
122 var mk = 1;
123
124 // Populate random seed array.
125 for (int i = 1; i < 55; i++)
126 {
127 var index = 21 * i % 55;
128 Seeds[index] = mk;
129 mk = Mod(mj - mk);
130 mj = Seeds[index];
131 }
132
133 // Generate random numbers to reduce seed bias.
134 for (int k = 1; k < 5; k++)
135 {
136 for (int i = 1; i < 56; i++)
137 {
138 Seeds[i] = Mod(Seeds[i] - Seeds[1 + (i + 30) % 55]);
139 }
140 }
141 }

147 public int Next()
148 {
149 var i = WrapIndex(Position1 + 1);

151 var next = Mod(Seeds[i] - Seeds[j]);
152
153 if (next == int.MaxValue)
154 next--;
155
156 Seeds[i] = next;
157 Position1 = i;
158 Position2 = j;
159 return next;
